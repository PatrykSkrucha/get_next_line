int check_for_nl(char *str)
{
	int len;

	if (!str)
		return (-3);
	len = 0;
	while(str[len])
	{	
		if (str[len] == '\n' && len + 1 == ft_strlen(str))
		{
			return (-2);
		}
		if (str[len] == '\n')
			return (len);
		len++;
	}
	return (-1);
}
